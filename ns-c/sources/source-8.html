179      * Computes resolution of nanoTime in worst case scenario
180      *
181      * According to https://docs.oracle.com/javase/7/docs/api/java/lang/System.html#nanoTime(),
182      * resolution of nanoTime is at least as good as that of currentTimeMillis()
183      *
184      * The algorithm to compute currentTimeMillis() resolution is adopted from
185      * https://www.javaworld.com/article/2077327/core-java/my-kingdom-for-a-good-timer.html
186      */
187     private static double compute_nanoTime_resolution() {
188         // JIT/hotspot warmup:
189         for (int r = 0; r < 3000; ++r) {
190             System.currentTimeMillis();
191         }
192 
193         long time = System.currentTimeMillis();
194         long time_prev = time;
195         long total_delta = 0;
196         int iteration = 5;
197 
198         for (int i = 0; i < iteration; ++i) {
199             // Busy wait until system time changes:
200             while (time == time_prev) {
201                 time = System.currentTimeMillis();
202             }
203             total_delta += (time - time_prev);
204             time_prev = time;
205         }
206 
207         return (total_delta / iteration) / 1000.0; // returns resolution in seconds
208     }
