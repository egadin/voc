74     private ByteBuffer loadResource(URL url) throws IOException {
75         InputStream stream = null;
76         try {
77             stream = url.openStream();
78             int initialBufferCapacity = Math.min(0x40000, stream.available() + 1);
79             if (initialBufferCapacity <= 2) {
80                 initialBufferCapacity = 0x10000;
81             } else {
82                 initialBufferCapacity = Math.max(initialBufferCapacity, 0x200);
83             }
84             ByteBuffer buf = ByteBuffer.allocate(initialBufferCapacity);
85             while (true) {
86                 if (!buf.hasRemaining()) {
87                     ByteBuffer newBuf = ByteBuffer.allocate(2 * buf.capacity());
88                     buf.flip();
89                     newBuf.put(buf);
90                     buf = newBuf;
91                 }
92                 int len = stream.read(buf.array(), buf.position(), buf.remaining());
93                 if (len <= 0) {
94                     break;
95                 }
96                 buf.position(buf.position() + len);
97             }
98             buf.flip();
99             return buf;
100         } finally {
101             if (stream != null) {
102                 stream.close();
103             }
104         }
